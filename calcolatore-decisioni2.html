<script>

    const state = {

        decision: ''

    };


    const decisionInput = document.getElementById('decision-input');

    const areaButtons = document.getElementById('area-buttons');


    decisionInput.addEventListener('input', (e) => {

        state.decision = e.target.value;

        if (state.decision.trim() !== '') {

            areaButtons.style.display = 'block';

        } else {

            areaButtons.style.display = 'none';

        }

    });


    function showView(viewId) {

        document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));

        document.getElementById(viewId).classList.add('active');

        

        // Update decision text in the new view

        document.getElementById('decision-display-financial').textContent = state.decision;

        document.getElementById('decision-display-quality').textContent = state.decision;

        document.getElementById('decision-display-time').textContent = state.decision;

    }


    function resetAndGoHome() {

        // Reset inputs

        decisionInput.value = '';

        document.getElementById('investment-input').value = '';

        document.getElementById('hours-input').value = '';

        document.getElementById('lifeWith').value = '';

        document.getElementById('lifeWithout').value = '';

        

        // Reset state

        state.decision = '';


        // Hide results and show calculators

        document.getElementById('financial-results').classList.add('hidden');

        document.getElementById('financial-calculator').style.display = 'block';

        document.getElementById('time-results').classList.add('hidden');

        document.getElementById('time-calculator').style.display = 'block';


        // Hide area buttons

        areaButtons.style.display = 'none';

        

        showView('start-view');

    }


    function formatCurrency(value) {

        return value.toLocaleString('it-IT', { style: 'currency', currency: 'EUR' });

    }


    function calculateFinancials() {

        const principal = parseFloat(document.getElementById('investment-input').value);

        if (isNaN(principal) || principal <= 0) {

            alert("Per favore, inserisci un importo valido.");

            return;

        }

        const ANNUAL_RETURN_RATE = 0.03;

        const valueIn20Years = principal * Math.pow(1 + ANNUAL_RETURN_RATE, 20);

        const valueIn18Years = principal * Math.pow(1 + ANNUAL_RETURN_RATE, 18);

        const costOfDelay = valueIn20Years - valueIn18Years;


        const resultsContainer = document.getElementById('financial-results');

        resultsContainer.innerHTML = `

            <div class="grid grid-cols-1 md:grid-cols-2 gap-4 text-center">

                <div class="bg-emerald-50 p-4 rounded-lg">

                    <p class="text-sm text-emerald-700">Valore tra 20 anni (agendo oggi)</p>

                    <p class="text-3xl font-bold text-emerald-800">${formatCurrency(valueIn20Years)}</p>

                </div>

                <div class="bg-red-50 p-4 rounded-lg">

                    <p class="text-sm text-red-700">Valore tra 20 anni (rimandando di 2 anni)</p>

                    <p class="text-3xl font-bold text-red-800">${formatCurrency(valueIn18Years)}</p>

                </div>

            </div>

            <div class="bg-yellow-100 border-l-4 border-yellow-500 text-yellow-800 p-4 rounded-r-lg">

                <h3 class="font-bold">Il costo del rinvio di 24 mesi è:</h3>

                <p class="text-2xl font-bold">- ${formatCurrency(costOfDelay)}</p>

            </div>

            <button onclick="resetAndGoHome()" class="btn bg-slate-500 hover:bg-slate-600">Analizza un'altra decisione</button>

        `;

        resultsContainer.classList.remove('hidden');

        document.getElementById('financial-calculator').style.display = 'none';

    }


    function calculateTime() {

        const weeklyHours = parseFloat(document.getElementById('hours-input').value);

        if (isNaN(weeklyHours) || weeklyHours <= 0) {

            alert("Per favore, inserisci un numero di ore valido.");

            return;

        }

        const WEEKS_IN_YEAR = 52;

        const timeIn20Years = weeklyHours * WEEKS_IN_YEAR * 20;

        const timeIn18Years = weeklyHours * WEEKS_IN_YEAR * 18;

        const lostTime = timeIn20Years - timeIn18Years;


        const resultsContainer = document.getElementById('time-results');

        resultsContainer.innerHTML = `

            <div class="grid grid-cols-1 md:grid-cols-2 gap-4 text-center">

                <div class="bg-amber-50 p-4 rounded-lg">

                    <p class="text-sm text-amber-700">Tempo totale guadagnato in 20 anni</p>

                    <p class="text-3xl font-bold text-amber-800">${timeIn20Years.toLocaleString('it-IT')} ore</p>

                </div>

                <div class="bg-red-50 p-4 rounded-lg">

                    <p class="text-sm text-red-700">Tempo guadagnato se rimandi di 2 anni</p>

                    <p class="text-3xl font-bold text-red-800">${timeIn18Years.toLocaleString('it-IT')} ore</p>

                </div>

            </div>

            <div class="bg-yellow-100 border-l-4 border-yellow-500 text-yellow-800 p-4 rounded-r-lg text-center">

                <h3 class="font-bold text-lg">Rimandando, perdi <span class="text-2xl">${lostTime.toLocaleString('it-IT')}</span> ore di qualità.</h3>

                <p class="mt-2 text-xl">Ne vale la pena?</p>

            </div>

            <button onclick="resetAndGoHome()" class="btn bg-slate-500 hover:bg-slate-600">Analizza un'altra decisione</button>

        `;

        resultsContainer.classList.remove('hidden');

        document.getElementById('time-calculator').style.display = 'none';

    }

</script>
